Define spec for data model


(s/def :person/fname string?)
(s/def :person/lname string?)
(s/def :person/birthDate inst?)

(s/def :address/roadName string?)
(s/def :address/houseNo string?)
(s/def :address/postCode int?)
(s/def :address/city string?)
(s/def :address/country string?)

(s/def ::address (s/keys :req-un [:address/roadName
                                  :address/houseNo
                                  :address/postCode
                                  :address/city
                                  :address/country]))

(s/def ::person (s/keys :req-un [:person/fname
                                 :person/lname
                                 :person/birthDate]
                        :opt-un [::address]))


(s/def :score/value int?)
(s/def ::score (s/keys :req-un [:score/value]))

Write clojure test case to call java API 


(def gson (Gson.))

(defn from-json [v t]
  (.fromJson gson v t))

(defn to-jtype [t v]
  (-> v
      (cheshire/generate-string)
      (from-json t)))

(defn get-score [v]
  (->> (u/to-jtype Person v)
       (.getScore (ScoreService.))
       (u/from-jtype)))

(s/fdef get-score
        :args (s/cat :v ::person)
        :ret ::score)

(ct/defspec get-score-test
            10
            (prop/for-all [v (s/gen ::person)]
                          (let [w (get-score v)]
                            (not= nil? w))))
